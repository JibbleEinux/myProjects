#include <stdio.h>
void display(int n, int a)
{
    if (n == 0)
    {
        printf("0");
    }
    else
    {

        if (a == n)
        {
        }
        else
        {
            printf("%d ", a);
            a++;
            display(n, a);
        }
    }
}
void revDisplay(int n)
{
    if (n > 0)
    {
        printf("%d ", n);
        revDisplay(n - 1);
    }
}

int main()
{
    int n, a = 1;
    scanf("%d", &n);
    display(n, a);
    revDisplay(n);
    return 0;
}

#include <stdio.h>
#include <math.h>

int polySeries(int n, int b)
{
    if (b == 1)
    {
        return 1;
    }
    else if (b == 2)
    {
        return n + polySeries(n, b - 1);
    }
    else
    {
        b--;
        return pow(n, b) + polySeries(n, b);
    }
}

int main()
{
    int n, m, result = 0;
    printf("Enter an Integer: ");
    scanf("%d", &n);
    m = n;
    result = polySeries(n, m);
    printf("Result is: %d\n", result);
    return 0;
}

#include <stdio.h>
void display(int array[], int n, int j, int p)
{
    if (n != 2)
    {
        if (n == 1)
        {
            printf("%d %d\n", array[j/2], array[j/2]);
        }
        else
        {
            if (array[p] <= array[n-1])
            {
                printf("%d %d\n", array[p], array[n-1]);
            }
            p++;
            display(array, n-1, j, p);
        }
    }

}
void displayForTwo(int array[], int n, int j)
{

    if (n == 2)
    {
        printf("%d %d\n", array[j/2 -1], array[j/2]);
    }

}
int main()
{
    int n, i, j;
    scanf("%d", &n);
    int array[n];
    for (i = 0; i < n; i++)
    {
        scanf("%d", &array[i]);
    }
    j = n;
    int p = 0;
    if (n>2)
    {
        display(array, n, j, p);
    }
    else
    {
        displayForTwo(array, n, j);
    }
    return 0;
}

#include <stdio.h>

long int multNumbers(int n)
{
    if (n >= 1)
        return n * multNumbers(n - 1);
    else
        return 1;
}

int main()
{
    int n;
    printf("Enter a positive integer: ");
    scanf("%d", &n);
    printf("Factorial of %d = %ld", n, multNumbers(n));
    return 0;
}

#include <stdio.h>
int fiboNum(int n)
{
    if (n <= 1)
        return n;
    return 2 + fiboNum(n - 1) + fiboNum(n - 2);
}

int main()
{
    int n;
    printf("Enter an positive Integer: ");
    scanf("%d", &n);
    printf("%d position fibonacci number is: %d", n, fiboNum(n));
    return 0;
}

#include <stdio.h>

int gcdFinder(int n1, int n2)
{
    if (n2 != 0)
        return gcdFinder(n2, n1 % n2);
    else
        return n1;
}

int main()
{
    int n1, n2;
    scanf("%d %d", &n1, &n2);
    printf("%d", gcdFinder(n1, n2));
    return 0;
}
